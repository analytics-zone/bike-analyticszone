Critical Power & W' - គំរូប្រតិបត្តិការជិះកង់កម្រិតខ្ពស់

Master Critical Power (CP) និង W Prime (W') សម្រាប់ល្បឿនដ៏ប្រសើរ ការព្យាករណ៍អស់កម្លាំង និងយុទ្ធសាស្ត្រប្រណាំង។ គំរូដ៏រឹងមាំតាមបែបវិទ្យាសាស្ត្រសម្រាប់ដំណើរការជិះកង់។

🎯 គន្លឹះសំខាន់ៗ

  • អំណាចសំខាន់ (CP)គឺជាថាមពលដែលមាននិរន្តរភាពអតិបរិមាសម្រាប់រយៈពេលយូរ - រឹងមាំខាងវិទ្យាសាស្ត្រជាង FTP
  • W' (W Prime)គឺជាសមត្ថភាពការងារ anaerobic របស់អ្នកលើសពី CP ដែលវាស់វែងជាគីឡូជូល។
  • W'Balanceតាមដានការថយចុះពេលវេលាជាក់ស្តែង និងការស្ដារឡើងវិញនូវសមត្ថភាព anaerobic អំឡុងពេលជិះ
  • CP ≈ FTP + 5-10Wនៅក្នុងការអនុវត្ត ប៉ុន្តែ CP គឺបានមកពីគណិតវិទ្យាពីការខិតខំប្រឹងប្រែងច្រើន។
  • សំខាន់សម្រាប់ MTB និងការខិតខំប្រឹងប្រែងអថេរដែលជាកន្លែងដែលការគ្រប់គ្រងល្បឿន និងការកើនឡើងមានសារៈសំខាន់

តើអ្វីជាថាមពលរិះគន់?

Critical Power (CP) គឺជាអត្រាមេតាបូលីសខ្ពស់បំផុតដែលអាចទ្រទ្រង់បានដោយមិនអស់កម្លាំងសម្រាប់រយៈពេលបន្ត។ វាតំណាងឱ្យព្រំដែនរវាងការបំប្លែងសារជាតិ aerobic ប្រកបដោយនិរន្តរភាព និងការធ្វើលំហាត់ប្រាណគ្មាននិរន្តរភាពដែលទាមទារការរួមចំណែក anaerobic ។ មិនដូច FTP (ការប៉ាន់ប្រមាណ 1 ម៉ោងតែមួយ) CP ត្រូវបានបង្កើតតាមគណិតវិទ្យាពីការខិតខំប្រឹងប្រែងជាអតិបរមាក្នុងរយៈពេលខុសៗគ្នា ដែលធ្វើឱ្យវាកាន់តែរឹងមាំ និងមានសុពលភាពតាមបែបវិទ្យាសាស្ត្រ។

វិទ្យាសាស្រ្តនៅពីក្រោយអំណាចសំខាន់

ទ្រឹស្ដីថាមពលដ៏សំខាន់បានកើតចេញពីការស្រាវជ្រាវសរីរវិទ្យានៃការធ្វើលំហាត់ប្រាណក្នុងទសវត្សរ៍ឆ្នាំ 1960 ហើយត្រូវបានកែលម្អសម្រាប់ការជិះកង់ក្នុងទសវត្សរ៍ឆ្នាំ 1990 ។ គំរូគឺផ្អែកលើទំនាក់ទំនងរយៈពេលនៃថាមពលអ៊ីពែរបូល:

ទំនាក់ទំនងរយៈពេលថាមពល

t = W '/ (P - CP)

កន្លែងណា៖

  • t= ដល់ពេលហត់នឿយ
  • ទំ= ទិន្នផលថាមពល
  • CP= ថាមពលសំខាន់ (វ៉ាត់)
  • W'= សមត្ថភាពការងារ anaerobic (គីឡូស៊ូល)

តើនេះមានន័យយ៉ាងណា៖នៅថាមពលណាមួយខាងលើ CP អ្នកមានការងារកំណត់ (W') មុនពេលហត់នឿយ។ នៅ CP ខ្លួនឯងអ្នកអាចបន្តតាមទ្រឹស្តីដោយគ្មានកំណត់។ នៅក្រោម CP អ្នកមិនដែលចូលទៅក្នុង W' ហើយអាចរក្សាការខិតខំប្រឹងប្រែងសម្រាប់រយៈពេលដ៏យូរ។

📚 មូលនិធិស្រាវជ្រាវ

ថាមពលរិះគន់ត្រូវបានគាំទ្រដោយការស្រាវជ្រាវរាប់ទសវត្សរ៍ដែលពិនិត្យដោយមិត្តភ័ក្តិ៖

  • Jones et al ។ (2019)៖"អំណាចសំខាន់៖ ទ្រឹស្តី និងកម្មវិធី" - ការត្រួតពិនិត្យយ៉ាងទូលំទូលាយនៅក្នុងទិនានុប្បវត្តិនៃសរីរវិទ្យាអនុវត្ត
  • Poole et al ។ (2016)៖"ថាមពលសំខាន់៖ កម្រិតនៃភាពអស់កម្លាំងដ៏សំខាន់" - ធ្វើឱ្យ CP មានសុពលភាពជាកម្រិតសរីរវិទ្យា
  • Vanhatalo et al ។ (២០១១)៖បង្ហាញពីការតម្រឹម CP ជាមួយនឹងស្ថានភាពស្ថិរភាព lactate អតិបរមា

ថាមពលសំខាន់ធៀបនឹង FTP៖ ភាពខុសគ្នាសំខាន់ៗ

FTP vs Critical Power (CP)៖ តើមួយណាល្អជាងសម្រាប់ Pacing?: ការស្វែងយល់ពីភាពខុសគ្នាយ៉ាងជ្រាលជ្រៅរវាងថាមពលកម្រិតមុខងារ និងគំរូថាមពលសំខាន់ (CP + W') ហើយមូលហេតុដែលក្រោយមកទៀតកំពុងធ្វើបដិវត្តន៍យុទ្ធសាស្ត្រល្បឿន។

ថាមពលកម្រិតមុខងារ (FTP)

និយមន័យ៖ថាមពលអតិបរមាមាននិរន្តរភាពប្រហែល 1 ម៉ោង។

ការធ្វើតេស្ត៖ការប្រឹងប្រែងតែមួយរយៈពេល 20 នាទី ឬ 60 នាទី។

ការគណនា៖FTP = 95% នៃថាមពល 20 នាទី (ឬ 100% នៃថាមពល 60 នាទី) ។

គុណសម្បត្តិ៖

  • សាមញ្ញ​ដើម្បី​សាកល្បង​និង​យល់​
  • ត្រូវការការខិតខំប្រឹងប្រែងតែមួយ
  • ស្តង់ដារឧស្សាហកម្មប្រើប្រាស់យ៉ាងទូលំទូលាយ
  • រួមបញ្ចូលនៅក្នុង TrainingPeaks, Zwift ជាដើម។

គុណវិបត្តិ៖

  • ការប៉ាន់ស្មានចំណុចតែមួយ (មិនសូវរឹងមាំ)
  • ផ្លូវចិត្តទាមទារការខិតខំប្រឹងប្រែង 20-60 នាទី។
  • កំហុសក្នុងល្បឿនប៉ះពាល់ដល់ភាពត្រឹមត្រូវ
  • គ្មានរង្វាស់សមត្ថភាព anaerobic

អំណាចសំខាន់ (CP)

និយមន័យ៖ថាមពលប្រកបដោយនិរន្តរភាពអតិបរមាសម្រាប់រយៈពេលគ្មានកំណត់តាមទ្រឹស្តី។

ការធ្វើតេស្ត៖កិច្ចខិតខំប្រឹងប្រែងអតិបរមាជាច្រើន (3-7 នាទី, 12 នាទី, 20 នាទីធម្មតា) ។

ការគណនា៖ខ្សែកោងគណិតវិទ្យាសមពីចំណុចទិន្នន័យច្រើន។

គុណសម្បត្តិ៖

  • រឹងមាំខាងវិទ្យាសាស្ត្រ (ការប្រឹងប្រែងច្រើន)
  • រួមបញ្ចូល W' (សមត្ថភាព anaerobic)
  • ភាពត្រឹមត្រូវប្រសើរជាងការធ្វើតេស្ត FTP តែមួយ
  • បើកដំណើរការតាមដាន W' Balance

គុណវិបត្តិ៖

  • តម្រូវឱ្យមានការខិតខំប្រឹងប្រែងអតិបរមា 3-5 ដាច់ដោយឡែកពីគ្នា។
  • កាន់តែស្មុគស្មាញក្នុងការគណនា
  • មិនសូវយល់ទូលំទូលាយ
  • ពិធីការសាកល្បងទាមទារកាន់តែច្រើន

🔍 នៅក្នុងការអនុវត្ត៖ CP ≈ FTP + 5-10W

សម្រាប់អ្នកជិះកង់ដែលបានទទួលការបណ្តុះបណ្តាលយ៉ាងល្អ Critical Power គឺជាធម្មតា5-10 វ៉ាត់ខ្ពស់ជាង FTP. ឧទាហរណ៍៖

  • ការពារ 10X៖250W (ពីការធ្វើតេស្ត 20 នាទី)
  • CP៖257W (ពីការធ្វើតេស្ត 3 នាទី, 12 នាទី, 20 នាទី)

CP តំណាងឱ្យកម្រិតទ្រឹស្ដីរយៈពេលគ្មានកំណត់ ខណៈពេលដែល FTP គឺជាការប៉ាន់ប្រមាណថាមពល 1 ម៉ោង។ ទាំងពីរមានប្រយោជន៍—FTP សម្រាប់ភាពសាមញ្ញ CP សម្រាប់ភាពជាក់លាក់ និងការតាមដាន W'។

តើ W' (W Prime) ជាអ្វី?

W' (បញ្ចេញសំឡេងថា "W Prime") គឺជាចំនួនកំណត់នៃការងារដែលអ្នកអាចអនុវត្តលើសពីថាមពលសំខាន់។គិតថាវាជា "ថ្ម anaerobic" របស់អ្នក ដែលជាកន្លែងផ្ទុកថាមពលមានកំណត់ ដែលបន្ថយនៅពេលជិះពីលើ CP និងងើបឡើងវិញយឺតពេលជិះក្រោម CP ។

W' និយមន័យ

W ' = (P - CP) × t

កន្លែងណា៖

  • W'= សមត្ថភាពការងារ anaerobic (គីឡូស៊ូល)
  • ទំ= ទិន្នផលថាមពល (វ៉ាត់)
  • CP= ថាមពលសំខាន់ (វ៉ាត់)
  • t= ពេលវេលាហត់នឿយ (វិនាទី)

ការប្រណាំង Off-road ពាក់ព័ន្ធនឹងការកើនឡើងថាមពលថេរខាងលើ CP ។ ប្រើ W'bal ដើម្បីគ្រប់គ្រង "ការផ្គូផ្គងដែលបានដុត"—ការកើនឡើងនីមួយៗធ្វើឱ្យ W' ហើយរយៈពេលនៃការស្តារឡើងវិញត្រូវតែគ្រប់គ្រាន់ដើម្បីស្តារសមត្ថភាពឡើងវិញ។

ការគ្រប់គ្រងការកើនឡើង MTB៖

  • មុនពេលវាយប្រហារ៖ពិនិត្យ W'bal ≥ 60% (ទុនបម្រុងគ្រប់គ្រាន់)
  • កំឡុងពេលកើនឡើង៖ទទួលយក W' depletion ប៉ុន្តែដឹងពីតម្លៃ
  • បន្ទាប់ពីការកើនឡើង៖ទម្លាក់ចុះក្រោម CP ដើម្បីស្ដារ W' មុនពេលការខិតខំប្រឹងប្រែងបន្ទាប់
  • ការប្រណាំងយឺត៖ប្រសិនបើ W'bal < 30% ជៀសវាងការកើនឡើងដ៏ធំ អ្នកនឹងផ្ទុះឡើង

3. លក្ខណៈវិនិច្ឆ័យ និងយុទ្ធសាស្ត្រប្រណាំងផ្លូវ

លក្ខណៈវិនិច្ឆ័យតម្រូវឱ្យមានការបង្កើនល្បឿន និងការវាយប្រហារម្តងហើយម្តងទៀត។ W'bal ជួយអ្នកឱ្យដឹងពីពេលដែលអ្នកអាចឆ្លើយតបនឹងការវាយប្រហារ និងពេលណាដែលអ្នកត្រូវការអនុញ្ញាតឱ្យសម្រាក។

ការសម្រេចចិត្តវាយលុក៖

  • W'bal = 85%៖ទៅជាមួយការសម្រាក - អ្នកមានទុនបំរុង
  • W'bal = 40%៖ប្រថុយប្រថាន - អាចផ្ទុះការដេញតាម
  • W'bal = 15%៖ស្នាក់នៅក្នុងកញ្ចប់ សង្គ្រោះ W' សម្រាប់ការរត់ចុងក្រោយ

4. ការរចនាបណ្ដុះបណ្ដាលចន្លោះពេល

ប្រើ CP និង W' ដើម្បីរៀបចំចន្លោះពេលយ៉ាងជាក់លាក់។ សម្រាប់ចន្លោះពេល VO₂max ថាមពលគួរតែជា CP + (W' / interval duration)។

ឧទាហរណ៍៖ ចន្លោះពេល VO₂max 5 នាទី។

  • CP = 250W, W' = 20 kJ
  • ថាមពលគោលដៅ៖250W + (20,000J / 300s) = 250W + 67W =317 វ៉
  • វាធ្វើអោយ W' អស់ទាំងស្រុងក្នុងរយៈពេល 5 នាទី។
  • ការងើបឡើងវិញ៖5-10 នាទីខាងក្រោម CP ដើម្បីស្ដារ W'

5. ការសាកល្បងពេលវេលា

សម្រាប់ការសាកល្បងពេលវេលាយូរជាង 30 នាទី សូមជិះក្រោម CP បន្តិច ដើម្បីជៀសវាង W' depletion ។ រក្សាទុក W សម្រាប់ការបញ្ចប់ការកើនឡើង។

40K TT Pacing៖

  • 35K ដំបូង៖95-98% នៃ CP (អភិរក្ស W')
  • 5K ចុងក្រោយ៖បង្កើនបន្តិចម្តងៗដល់ CP + 10-20W (ប្រើ W')
  • 1K ចុងក្រោយ៖ទទេ W' ទាំងស្រុង (បញ្ចប់ការរត់)

Critical Power & W': សំណួរដែលគេសួរញឹកញាប់

តើ CP ល្អជាង FTP ទេ?

CP មានភាពរឹងមាំខាងវិទ្យាសាស្ត្រព្រោះ​វា​បាន​មក​ពី​ការ​ខិត​ខំ​ច្រើន​មិន​មែន​ការ​ធ្វើ​តេស្ត​តែ​មួយ​ទេ។ ទោះយ៉ាងណាក៏ដោយ FTP គឺសាមញ្ញជាង និងយល់យ៉ាងទូលំទូលាយ។ សម្រាប់អ្នកជិះកង់ភាគច្រើន FTP គឺគ្រប់គ្រាន់ហើយ។ ប្រើ CP ប្រសិនបើអ្នកចង់បានភាពជាក់លាក់ ការតាមដាន W ឬគំរូល្បឿនប្រណាំង។ នៅក្នុងការអនុវត្ត CP ≈ FTP + 5-10W ។

តើ W' ត្រូវការពេលប៉ុន្មានដើម្បីជាសះស្បើយពេញលេញ?

ការងើបឡើងវិញគឺអិចស្ប៉ូណង់ស្យែល មិនមែនលីនេអ៊ែរទេ។នៅ 100W ក្រោម CP W' ងើបឡើងវិញ ~ 50% ក្នុងរយៈពេល 6 នាទី 75% ក្នុងរយៈពេល 12 នាទី 90% ក្នុងរយៈពេល 20 នាទី។ ការងើបឡើងវិញពេញលេញ (99%+) ចំណាយពេល 30-40 នាទីនៅថាមពលទាបបំផុត។ កាន់តែខាងក្រោម CP ដែលអ្នកជិះ នោះ W' កាន់តែលឿន។ ការសម្រាកពេញលេញ (0W) គឺមិនល្អបំផុតទេ - ការបង្វិលពន្លឺ (~100-150W) បង្កើនល្បឿននៃការស្តារឡើងវិញ។

តើខ្ញុំអាចហ្វឹកហាត់ដើម្បីបង្កើន W បានទេ?

បាទ-W' អាចហ្វឹកហាត់បានតាមរយៈចន្លោះពេល anaerobic ។ចន្លោះពេល VO₂max (3-8 នាទីនៅ 110-120% CP) និងការកើនឡើងម្តងហើយម្តងទៀត (30-90s នៅ 150%+ CP) ពង្រីក W' ។ អ្នករត់ប្រណាំងតាមដងផ្លូវ និងអ្នកប្រណាំង MTB ជាធម្មតាមាន W' ខ្ពស់ជាងអ្នកជិះដែលមានការស៊ូទ្រាំ។ ការផ្តោតអារម្មណ៍លើការបណ្តុះបណ្តាល៖ វគ្គ 1-2 ​​ក្នុងមួយសប្តាហ៍នៃចន្លោះពេលអាំងតង់ស៊ីតេខ្ពស់ដែលធ្វើអោយ W' ថយចុះទាំងស្រុង បន្ទាប់មកដោយការស្តារឡើងវិញគ្រប់គ្រាន់។

តើ W' ថយចុះនៅពេលជិះរីកចម្រើនទេ?

បាទ-សមត្ថភាពរបស់ W ថយចុះជាមួយនឹងភាពអស់កម្លាំងបង្គរ។នៅពេលជិះដំបូង អ្នកប្រហែលជាមាន 20 kJ ដែលអាចប្រើបាន។ បន្ទាប់ពី 2-3 ម៉ោងនៃការជិះខ្លាំង ប្រសិទ្ធភាព W' អាចធ្លាក់ចុះដល់ 12-15 kJ ។ នេះ​ជា​មូលហេតុ​ដែល​ការ​កើន​ឡើង​នៃ​ការ​ប្រណាំង​ចុង​ក្រោយ​មាន​អារម្មណ៍​ថា​ពិបាក​ជាង​មុន—ទុន​បម្រុង anaerobic របស់​អ្នក​ត្រូវ​បាន​គេ​សម្រប​សម្រួល។ Bike Analytics អាចយកគំរូតាម W' depletion ដោយប្រើកត្តាអស់កម្លាំង។

តើខ្ញុំគួរធ្វើតេស្ត CP និង W ឡើងវិញញឹកញាប់ប៉ុណ្ណា?

រៀងរាល់ 8-12 សប្តាហ៍កំឡុងពេលដំណើរការបណ្តុះបណ្តាល។ CP កើនឡើងយឺតជាង FTP (វាមានស្ថេរភាពជាង)។ W 'អាចផ្លាស់ប្តូរយ៉ាងខ្លាំងជាមួយនឹងការបណ្តុះបណ្តាល anaerobic គោលដៅ។ ធ្វើតេស្តឡើងវិញបន្ទាប់ពីការហ្វឹកហាត់សំខាន់ៗ ជំងឺ ឬរបួស។ CP/W' មានភាពធន់នឹងការប្រែប្រួលនៃកាយសម្បទារយៈពេលខ្លីជាង FTP។

តើខ្ញុំអាចប្រើ CP សម្រាប់ការបណ្តុះបណ្តាលក្នុងផ្ទះបានទេ?

ពិតប្រាកដណាស់ - CP គឺល្អសម្រាប់ការបណ្តុះបណ្តាលក្នុងផ្ទះ។គ្រូបណ្តុះបណ្តាលឆ្លាតវៃផ្តល់នូវថាមពលមានស្ថេរភាព ធ្វើឱ្យការធ្វើតេស្ត CP មានភាពត្រឹមត្រូវខ្ពស់។ Zwift, TrainerRoad និងវេទិកាផ្សេងទៀតគាំទ្រការហាត់ប្រាណដែលមានមូលដ្ឋានលើ CP ។ ការតាមដាន W'bal ដំណើរការយ៉ាងល្អឥតខ្ចោះនៅក្នុងផ្ទះដែលថាមពលគឺស្រប។ គ្រូបង្វឹកជាច្រើនចូលចិត្ត CP សម្រាប់ការបណ្តុះបណ្តាលក្នុងផ្ទះដែលមានរចនាសម្ព័ន្ធ។

តើអ្វីជាគំរូ Skiba W' Balance?

គំរូឆ្នាំ 2012 របស់បណ្ឌិត Philip Skiba តាមគណិតសាស្ត្រតាមដានការបំផ្លិចបំផ្លាញ និងការស្តារឡើងវិញ។វាប្រើសមីការឌីផេរ៉ង់ស្យែលជាមួយនឹងពេលវេលាថេរτ ≈ 377 វិនាទីសម្រាប់ kinetics ការងើបឡើងវិញ។ គំរូនេះត្រូវបានអនុវត្តនៅក្នុង WKO5, Golden Cheetah និង Bike Analytics ។ វាជាស្តង់ដារមាសសម្រាប់ការគណនា W'bal ពេលវេលាពិតអំឡុងពេលជិះ។ ការស្រាវជ្រាវ៖ Skiba et al ។ (2012, 2014, 2021) ផ្នែកវេជ្ជសាស្ត្រ និងវិទ្យាសាស្ត្រក្នុងកីឡា និងលំហាត់ប្រាណ។

តើ CP និង W អាចទស្សន៍ទាយការប្រណាំងបានទេ?

បាទ - ជាមួយនឹងភាពត្រឹមត្រូវខ្ពស់សម្រាប់ការខិតខំប្រឹងប្រែង 3-60 នាទី។គំរូ CP អាចទស្សន៍ទាយពីពេលវេលាដល់ការហត់នឿយនៅថាមពលដែលបានផ្តល់ឱ្យណាមួយ។ ឧទាហរណ៍៖ ប្រសិនបើ CP = 250W និង W' = 20 kJ អ្នកអាចទ្រទ្រង់ 300W ក្នុងរយៈពេល 6.67 នាទី (20,000J / 50W = 400s)។ សម្រាប់ការខិតខំប្រឹងប្រែងយូរជាងនេះ (> 60 នាទី) CP ប៉ាន់ស្មានថាមពលប្រកបដោយនិរន្តរភាពបន្តិច ដោយសារកត្តាអស់កម្លាំងបន្ថែម។

តើកម្ពស់ប៉ះពាល់ដល់ CP និង W យ៉ាងដូចម្តេច?

CP ថយចុះ ~ 1% ក្នុង 300 ម៉ែត្រពីលើនីវ៉ូទឹក 1500m ។W' មិនសូវប៉ះពាល់ទេ ដោយសារសមត្ថភាព anaerobic មិនអាស្រ័យលើអុកស៊ីសែន។ នៅ 2500m រំពឹងថា CP នឹងធ្លាក់ចុះ 3-4% ប៉ុន្តែ W' នៅតែស្រដៀងគ្នា។ នេះមានន័យថានៅកម្ពស់ ថាមពលប្រកបដោយនិរន្តរភាពរបស់អ្នកធ្លាក់ចុះ ប៉ុន្តែសមត្ថភាពកើនឡើងរបស់អ្នក (ទាក់ទងទៅនឹង CP ថ្មី) ត្រូវបានរក្សា។ សាកល្បង CP ឡើងវិញនៅរយៈកម្ពស់សម្រាប់តំបន់បណ្តុះបណ្តាលត្រឹមត្រូវ។

តើខ្ញុំគួរប្រើ CP សម្រាប់ការបណ្តុះបណ្តាលកម្រិត 4 ទេ?

បាទ-CP កំណត់កម្រិតតំបន់ 4 យ៉ាងជាក់លាក់។ចន្លោះពេលកម្រិតគួរតែស្ថិតនៅ 95-105% នៃ CP ។ មិនដូច FTP (ប៉ាន់ស្មានពីការធ្វើតេស្តតែមួយ) CP ផ្តល់នូវកម្រិតដែលបានមកពីគណិតវិទ្យា។ សម្រាប់​ការ​ប្រឹងប្រែង​ដែល​មាន​និរន្តរភាព (2×20 នាទី) ជិះ​នៅ CP ។ សម្រាប់ចន្លោះពេលខ្លីជាង (5×5 នាទី) ជិះនៅ CP + 3-5% ។ Bike Analytics គណនាតំបន់បណ្តុះបណ្តាលពី CP ដោយស្វ័យប្រវត្តិ.
