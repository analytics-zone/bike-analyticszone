  let wbal = Wprime * 1000; // Convert to joules
  const wbalHistory = [];

  for (let i = 1; i < powerData.length; i++) {
    const dt = powerData[i].time - powerData[i-1].time;
    const power = powerData[i].power;

    if (power > CP) {
      // Expenditure above CP
      const expenditure = (power - CP) * dt;
      wbal -= expenditure;
    } else {
      // Recovery below CP
      const deltaCP = CP - power;
      const tau = 546 * Math.exp(-0.01 * deltaCP) + 316;
      const recovery = (Wprime * 1000 - wbal) * (1 - Math.exp(-dt / tau));
      wbal += recovery;
    }

    // Ensure W'bal doesn't exceed max or go negative
    wbal = Math.max(0, Math.min(wbal, Wprime * 1000));

    wbalHistory.push({
      time: powerData[i].time,
      wbal: wbal / 1000, // kJ
      percent: (wbal / (Wprime * 1000)) * 100
    });
  }

  return wbalHistory;
}

9. VAM (VelocitГ  Ascensionale Media)

бһҡбһјбһ”бһҳбһ“бҹ’бһҸбҹ–

VAM (m/h) = бһҖбһ¶бһҡбһҖбһҫбһ“бһЎбһҫбһ„ (m) / бһҳбҹүбҹ„бһ„ (бһҳбҹүбҹ„бһ„)

VAM Benchmarksбҹ–

VAM (m/h)бһҖбһҳбҹ’бһҡбһ·бһҸбһ§бһ‘бһ¶бһ бһҡбһҺбҹҚ
бҹҰбҹ бҹ  - бҹ©бҹ бҹ бһҖбҹҶбһҹбһ¶бһ“бҹ’бһҸбһўбҹ’бһ“бһҖбһҮбһ·бҹҮбһҖбҹ’бһӣбһ№бһ”бһӣбһҫбһҖбһ¶бһҡбһЎбһҫбһ„бһ—бҹ’бһ“бҹҶбһҖбҹ’бһ“бһ»бһ„бһҹбҹ’бһҡбһ»бһҖ

1200 - 1500бһңбһҡбһҮбһ“ бһҹбҹ’бһҳбҹҗбһӮбҹ’бһҡбһ…бһ·бһҸбҹ’бһҸбһўбҹ’бһ“бһҖбһЎбһҫбһ„бһ—бҹ’бһ“бҹҶбһҗбҹ’бһ“бһ¶бһҖбҹӢбһҮбһ¶бһҸбһ·
1500 - 1800бһңбһ·бһҮбҹ’бһҮбһ¶бһҮбһёбһңбҹҲбһҠбҹҶбһҺбһҫбһҡбһҖбҹҶбһҹбһ¶бһ“бҹ’бһҸбһ–бһ·бһ—бһ–бһӣбҹ„бһҖбһҖбҹ’бһ“бһ»бһ„бһҹбҹ’бһҡбһ»бһҖ
> 1800бһўбҹ’бһ“бһҖбһҲбҹ’бһ“бҹҮбһҡбһ„бҹ’бһңбһ¶бһ“бҹӢбһ’бҹҶPogaДҚar, Vingegaard бһ“бҹ…бһӣбһҫбһҖбһ¶бһҡбһЎбһҫбһ„бһҹбҹҶбһҒбһ¶бһ“бҹӢбҹ—

бһҖбһ¶бһҡбһӮбһҺбһ“бһ¶бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚбҹ–

бһҹбҹҒбһҺбһ¶бһҡбһёбһҷбҹүбһјбҹ–бһЎбһҫбһ„бһ—бҹ’бһ“бҹҶ Alpe d'Huez

  • бһҖбһ¶бһҡбһҖбһҫбһ“бһЎбһҫбһ„бһҖбһҳбҹ’бһ–бһҹбҹӢ: 1100 бһҳбҹүбҹӮбһҸбҹ’бһҡ
  • бһ–бҹҒбһӣбһңбҹҒбһӣбһ¶: 55 бһ“бһ¶бһ‘бһё = 0.917 бһҳбҹүбҹ„бһ„бҹ”
  • VAM = 1100 / 0.917 =1200 бһҳбҹүбҹӮбһҸ / бһҳбҹүбҹ„бһ„бҹ”

бһҖбһ¶бһҡбһ”бһҖбһҹбҹ’бһҡбһ¶бһҷбҹ– бһҖбһ¶бһҡбһЎбһҫбһ„бһҖбһҳбҹ’бһҡбһ·бһҸбһ”бҹ’бһҡбһҖбһҪбһҸбһ”бҹ’бһҡбһҮбҹӮбһ„

}

10. бһҖбһ¶бһҡбһ”бҹүбһ¶бһ“бҹӢбһҹбҹ’бһҳбһ¶бһ“ VAM бһ‘бҹ… W/kg

бһҡбһјбһ”бһҳбһ“бҹ’бһҸбҹ–

W/kg вүҲ VAM (m/h) / 100 / (бһҮбһҳбҹ’бһҡбһ¶бһӣ% + 3)

бһҖбһ¶бһҡбһӮбһҺбһ“бһ¶бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚбҹ–

бһҹбҹҒбһҺбһ¶бһҡбһёбһҷбҹүбһјбҹ–бһЎбһҫбһ„бһҮбһ¶бһҳбһҪбһҷбһ“бһ№бһ„бһҮбһҳбҹ’бһҡбһ¶бһӣбһҳбһ’бҹ’бһҷбһҳ 8% VAM = 1200 m/h

W / бһӮбһёбһЎбһјбһҖбҹ’бһҡбһ¶бһҳ = 1200 / 100 / (8 + 3)

W/kg = 12/11 =4.36 бһңбҹү / бһӮбһёбһЎбһјбһҖбҹ’бһҡбһ¶бһҳ

бһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҶбҹ’бһӣбһ„бһҖбһ¶бһҸбҹӢбҹ– бһҮбһ¶бһҳбһҪбһҷбһ“бһ№бһ„бһўбҹ’бһ“бһҖбһҮбһ·бҹҮ 70 бһӮбһёбһЎбһјбһҖбҹ’бһҡбһ¶бһҳ вҶ’ бһҗбһ¶бһҳбһ–бһӣ 305W бһ‘бҹ’бһҡбһ‘бҹ’бһҡбһ„бҹӢбһ“бҹ…бһ–бҹҒбһӣбһЎбһҫбһ„

function estimateWkgFromVAM(vam, gradientPercent) {
  return (vam / 100 / (gradientPercent + 3)).toFixed(2);
}

11. бһҸбһҳбҹ’бһҡбһјбһңбһҖбһ¶бһҡбһҗбһ¶бһҳбһ–бһӣбһҹбһҡбһ»бһ”

бһҡбһјбһ”бһҳбһ“бҹ’бһҸбһ–бҹҒбһүбһӣбҹҒбһүбҹ–

P_total = P_aero + P_gravity + P_rolling + P_kinetic

бһҡбһјбһ”бһҳбһ“бҹ’бһҸбһҹбһҳбһ¶бһҹбһ’бһ¶бһҸбһ»бҹ–

бһҖбһ¶бһҡбһўбһјбһҹбһҸбһ¶бһҳбһўбһ¶бһҖбһ¶бһҹбҹ–
P_aero = CdA Г— 0.5 Г— ПҒ Г— VВі
бһ‘бҹҶбһ“бһ¶бһү (бһЎбһҫбһ„бһ—бҹ’бһ“бҹҶ)бҹ–
P_gravity = m Г— g Г— sin(Оё) Г— V
бһ—бһ¶бһ–бһ’бһ“бҹӢбһ“бһ№бһ„бһҖбһ¶бһҡбһҡбҹҶбһҖбһ·бһӣбҹ–
P_rolling = Crr Г— m Г— g Г— cos (Оё) Г— V
Kinetic (бһҖбһ¶бһҡбһ”бһ„бҹ’бһҖбһҫбһ“бһӣбҹ’бһ”бһҝбһ“)бҹ–

бһўбһҗбҹҒбһҡ бһ“бһ·бһ„бһўбһҗбҹҒбһҡбҹ–

  • бһҹбҹҠбһёбһҢбһёбһўбҹҒ= бһҳбҹҒбһӮбһ»бһҺбһўбһјбһҹ Г— бһ•бҹ’бһ‘бҹғбһҒбһ¶бһ„бһҳбһ»бһҒ (mВІ)
    • бһҖбҹ’бһҡбһҺбһ¶бһҸбҹӢбһҖбһ„бҹӢбһ•бҹ’бһӣбһјбһңбһ’бһҳбҹ’бһҳбһҸбһ¶бҹ– 0.35-0.40 mВІ
    • бһ’бҹ’бһӣбһ¶бһҖбҹӢбһ…бһ»бҹҮ: 0.32-0.37 mВІ
    • бһ‘бһёбһҸбһ¶бҹҶбһ„ TT: 0.20-0.25 mВІ
  • ПҒ= бһҠбһ„бҹӢбһҹбҹҠбһёбһҸбҹҒбһҒбҹ’бһҷбһӣбҹӢ (1.225 бһӮбһёбһЎбһјбһҖбҹ’бһҡбһ¶бһҳ/mВі бһ“бҹ…бһҖбһҳбҹ’бһҡбһ·бһҸбһ‘бһ№бһҖбһҹбһҳбһ»бһ‘бҹ’бһҡ 15В°C)
  • бһң= бһӣбҹ’бһ”бһҝбһ“ (m/s)
  • бһҳ= бһҳбҹүбһ¶бһҹбһҹбһҡбһ»бһ” (бһўбҹ’бһ“бһҖбһҮбһ·бҹҮ + бһҖбһ„бҹӢ, бһӮбһёбһЎбһјбһҖбҹ’бһҡбһ¶бһҳ)
  • g= бһ‘бҹҶбһ“бһ¶бһү (9.81 m/sВІ)
  • Оё= бһҳбһ»бҹҶбһҮбһҳбҹ’бһҡбһ¶бһӣ (бһҡбҹүбһ¶бһҠбҹ’бһҷбһ„бҹӢ бһ¬бһҠбһәбһҖбҹ’бһҡбҹҒбһ”бҹҶбһ”бҹ’бһӣбҹӮбһ„)
  • Cr= бһҳбҹҒбһӮбһ»бһҺбһ’бһ“бҹӢбһ“бһ№бһ„бһҡбҹҶбһҖбһ·бһӣ (~0.004 бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҹбҹҶбһ”бһҖбһҖбһ„бҹӢбһ•бҹ’бһӣбһјбһңбһӣбҹ’бһў)
  • бһҖ= бһҖбһ¶бһҡбһ”бһ„бҹ’бһҖбһҫбһ“бһӣбҹ’бһ”бһҝбһ“ (m/sВІ)

бһҹбҹҒбһҺбһ¶бһҡбһёбһҷбҹүбһјбҹ–

  • бһӣбҹ’бһ”бһҝбһ“: 40 бһӮбһёбһЎбһјбһҳбҹүбҹӮбһҸбҹ’бһҡ / бһҳбҹүбҹ„бһ„ = 11.11 m / s

  • бһ‘бһҳбҹ’бһ„бһ“бҹӢбһҹбһҡбһ»бһ”бҹ– бҹ§бҹҘ бһӮбһёбһЎбһјбһҖбҹ’бһҡбһ¶бһҳ (бһўбҹ’бһ“бһҖбһҮбһ·бҹҮ) + бҹЁ бһӮбһёбһЎбһјбһҖбҹ’бһҡбһ¶бһҳ (бһҖбһ„бҹӢ) = бҹЁбҹЈ бһӮбһёбһЎбһјбһҖбҹ’бһҡбһ¶бһҳ
  • бһ•бҹ’бһӣбһјбһңбһҡбһ¶бһ”бһҹбҹ’бһҳбһҫ (бһҮбһҳбҹ’бһҡбһ¶бһӣ = 0В°)

бһҖбһ¶бһҡбһўбһ“бһ»бһңбһҸбҹ’бһҸ JavaScriptбҹ–

бһҳбһ»бһҒбһ„бһ¶бһҡбһӮбһҺбһ“бһ¶бһҗбһ¶бһҳбһ–бһӣбһҠбҹӮбһӣбһҸбҹ’бһҡбһјбһңбһҖбһ¶бһҡ (бһ”бҹүбһ¶бһҡбҹүбһ¶бһҳбҹүбҹӮбһҸбҹ’бһҡ) {

    бһӣбҹ’бһ”бһҝбһ“Kph,

    бһҳбҹүбһ¶бһҹбҹӢ = 83, // бһӮбһёбһЎбһјбһҖбҹ’бһҡбһ¶бһҳ (бһўбҹ’бһ“бһҖбһҮбһ·бҹҮ + бһҖбһ„бҹӢ)
    gradientPercent = 0, // %
    Crr = 0.004, // бһ—бһ¶бһ–бһ’бһ“бҹӢбһ“бһ№бһ„бһҡбҹҶбһҖбһ·бһӣ
    accelerationMps2 = 0 // m/sВІ
  } = бһ”бҹүбһ¶бһҡбҹүбһ¶бһҳбҹүбҹӮбһҸбҹ’бһҡ;

  // бһ”бҹҶбһ”бҹ’бһӣбҹӮбһ„бһӣбҹ’бһ”бһҝбһ“бһ‘бҹ…бһҮбһ¶ m/s

  const theta = Math.atan(gradientPercent / 100);

  const P_aero = CdA * 0.5 * rho * Math.pow(V, 3);
  const P_gravity = бһҳбҹүбһ¶бһҹбҹӢ * 9.81 * Math.sin(theta) * V;
  const P_rolling = Crr * бһҳбҹүбһ¶бһҹбҹӢ * 9.81 * Math.cos(theta) * V;
  const P_kinetic = бһҳбҹүбһ¶бһҹбҹӢ * accelerationMps2 * V;

  бһҸбҹ’бһҡбһЎбһ”бҹӢ {
    бһҹбһҡбһ»бһ”бҹ– Math.round(P_aero + P_gravity + P_rolling + P_kinetic),
    aero: Math.round(P_aero),
    бһ‘бҹҶбһ“бһ¶бһүбҹ– Math.round(P_gravity),
    бһҡбһҳбҹҖбһӣбҹ– Math.round(P_rolling),
    kineticбҹ– Math.round(P_kinetic)
  };
}
