គោលការណ៍ឯកជនភាពសម្រាប់ការវិភាគកង់

ធ្វើបច្ចុប្បន្នភាពចុងក្រោយ៖ថ្ងៃទី 10 ខែមករា ឆ្នាំ 2025 |មានប្រសិទ្ធភាព កាលបរិច្ឆេទ៖ថ្ងៃទី 10 ខែមករា ឆ្នាំ 2025

សេចក្តីផ្តើម

Bike Analytics ("យើង" "របស់យើង" ឬ "កម្មវិធី") ប្តេជ្ញាការពារភាពឯកជនរបស់អ្នក។ គោលការណ៍ឯកជនភាពនេះ។ ពន្យល់ពីរបៀបដែលកម្មវិធីទូរស័ព្ទរបស់យើង (iOS និង Android) ចូលប្រើ ប្រើប្រាស់ និងការពារទិន្នន័យសុខភាពពីអ្នក ឧបករណ៍។

គោលការណ៍ឯកជនភាពសំខាន់៖Bike Analytics ដំណើរការលើ កម៉ាស៊ីនមេសូន្យ, ក្នុងស្រុកតែប៉ុណ្ណោះ ស្ថាបត្យកម្ម. ទិន្នន័យសុខភាពទាំងអស់ត្រូវបានចូលប្រើពី Apple HealthKit (iOS) ឬ Health Connect (Android) នៅសល់តែលើឧបករណ៍រូបវន្តរបស់អ្នក ហើយមិនត្រូវបានបញ្ជូនទៅកាន់ម៉ាស៊ីនមេខាងក្រៅឡើយ សេវាពពក ឬភាគីទីបី។

1. ការចូលប្រើទិន្នន័យសុខភាព

Bike Analytics រួមបញ្ចូលជាមួយវេទិកាសុខភាពដើមរបស់ឧបករណ៍របស់អ្នក ដើម្បីផ្តល់នូវការហាត់ប្រាណជិះកង់លម្អិត ការវិភាគ៖

1.1 iOS - ការរួមបញ្ចូល Apple HealthKit

នៅលើឧបករណ៍ iOS Bike Analytics រួមបញ្ចូលជាមួយកញ្ចប់សុខភាព Appleដើម្បីចូលដំណើរការជិះកង់ ទិន្នន័យលំហាត់ប្រាណ។ យើងស្នើសុំការចូលប្រើបានតែអានទៅ៖

  • វគ្គហាត់ប្រាណ៖វគ្គហាត់ប្រាណជិះកង់ជាមួយនឹងពេលវេលា និងរយៈពេល
  • ចម្ងាយ៖ចម្ងាយជិះកង់សរុប
  • អត្រាបេះដូង៖ទិន្នន័យអត្រាបេះដូងអំឡុងពេលហាត់ប្រាណ
  • ថាមពលសកម្ម៖កាឡូរីត្រូវបានដុតកំឡុងពេលជិះកង់
  • ថាមពលជិះកង់៖ទិន្នន័យថាមពល (វ៉ាត់) សម្រាប់ការវិភាគដំណើរការ
  • ល្បឿនជិះកង់៖ទិន្នន័យល្បឿនសម្រាប់ការវិភាគល្បឿន

ការអនុលោមតាម HealthKit របស់ Apple៖Bike Analytics ប្រកាន់ខ្ជាប់នូវគោលការណ៍ណែនាំរបស់ Apple HealthKit ទាំងអស់។ ទិន្នន័យសុខភាពរបស់អ្នកត្រូវបានដំណើរការទាំងស្រុងនៅលើឧបករណ៍ iOS របស់អ្នក ហើយមិនដែលទុកវាចោលឡើយ។ យើងមិនដែលចែករំលែក HealthKit ទេ។ ទិន្នន័យជាមួយភាគីទីបី វេទិកាផ្សព្វផ្សាយពាណិជ្ជកម្ម ឬឈ្មួញកណ្តាលទិន្នន័យ។

1.2 ប្រព័ន្ធប្រតិបត្តិការ Android - ការរួមបញ្ចូលការតភ្ជាប់សុខភាព

ប្រភេទទិន្នន័យសុខភាព ការអនុញ្ញាត គោលបំណង
វគ្គលំហាត់ប្រាណ READ_EXERCISE ដើម្បីកំណត់អត្តសញ្ញាណ និងនាំចូលវគ្គហាត់ប្រាណជិះកង់ពី Health Connect
កំណត់ត្រាពីចម្ងាយ READ_DISTANCE ដើម្បីបង្ហាញចម្ងាយសរុបក្នុងមួយជិះ និងគណនាម៉ែត្រល្បឿន
កំណត់ត្រាអត្រាបេះដូង READ_HEART_RATE ដើម្បីបង្ហាញតារាងអត្រាបេះដូង គណនាអត្រាបេះដូងជាមធ្យម និងអតិបរមា
កំណត់ត្រាល្បឿន READ_SPEED ដើម្បីគណនា និងបង្ហាញតំបន់ល្បឿន និងល្បឿននៃការជិះកង់របស់អ្នក។
កំណត់ត្រាថាមពល READ_POWER ដើម្បីវិភាគទិន្នផលថាមពល (វ៉ាត់) និងគណនា FTP/TSS
កាឡូរីដុត READ_TOTAL_CALORIES_BURNED ដើម្បីផ្តល់ទិដ្ឋភាពទូទៅនៃការចំណាយថាមពលអំឡុងពេលជិះ

ការអនុញ្ញាត Android៖ការអនុញ្ញាតទាំងនេះត្រូវបានស្នើសុំក្នុងអំឡុងពេលចាប់ផ្តើមដំបូងរបស់កម្មវិធី។ អ្នក អាចដកហូតការអនុញ្ញាតទាំងនេះនៅពេលណាក៏បានតាមរយៈការកំណត់ Android → កម្មវិធី → Health Connect → Bike វិភាគ។

1.3 របៀបដែលយើងប្រើប្រាស់ទិន្នន័យសុខភាព

ទិន្នន័យសុខភាពទាំងអស់ត្រូវបានប្រើប្រាស់ទាំងស្រុងសម្រាប់គោលបំណងដូចខាងក្រោម:

  • ការបង្ហាញលំហាត់ប្រាណ៖បង្ហាញវគ្គជិះកង់របស់អ្នកជាមួយនឹងរង្វាស់លម្អិត (ចម្ងាយ, ពេលវេលា, ល្បឿន, ចង្វាក់បេះដូង, ថាមពល)
  • ការវិភាគការអនុវត្ត៖គណនាតំបន់ថាមពល FTP (ថាមពលកម្រិតមុខងារ) និង TSS (ពិន្ទុស្ត្រេសហ្វឹកហាត់)
  • ការតាមដានវឌ្ឍនភាព៖បង្ហាញនិន្នាការនៃការអនុវត្ត ភាពល្អបំផុតផ្ទាល់ខ្លួន និងការហាត់ប្រាណ សេចក្តីសង្ខេប
  • ការនាំចេញទិន្នន័យ៖អនុញ្ញាតឱ្យអ្នកនាំចេញទិន្នន័យការហាត់ប្រាណរបស់អ្នកទៅជាទម្រង់ CSV សម្រាប់ការប្រើប្រាស់ផ្ទាល់ខ្លួន

1.4 ការផ្ទុកទិន្នន័យ

🔒 ការធានាឯកជនភាពសំខាន់ៗ៖

ទិន្នន័យសុខភាពទាំងអស់នៅតែមានតែនៅលើឧបករណ៍រាងកាយរបស់អ្នក។

  • iOS៖ទិន្នន័យត្រូវបានរក្សាទុកដោយប្រើ iOS ទិន្នន័យស្នូល និង UserDefaults (នៅលើឧបករណ៍តែប៉ុណ្ណោះ)
  • ប្រព័ន្ធប្រតិបត្តិការ Android៖ទិន្នន័យត្រូវបានរក្សាទុកដោយប្រើមូលដ្ឋានទិន្នន័យ Android Room (SQLite នៅលើឧបករណ៍)
  • គ្មានទិន្នន័យត្រូវបានផ្ទុកឡើងទៅកាន់ម៉ាស៊ីនមេខាងក្រៅទេ។
  • គ្មានទិន្នន័យត្រូវបានបញ្ជូនតាមអ៊ីនធឺណិតទេ។
  • គ្មានការធ្វើសមកាលកម្មលើពពក ឬការបម្រុងទុកទិន្នន័យសុខភាពទេ។
  • គ្មានភាគីទីបីចូលប្រើទិន្នន័យសុខភាពរបស់អ្នកទេ។

ពេលវេលាតែមួយគត់ដែលទិន្នន័យចាកចេញពីឧបករណ៍របស់អ្នកគឺនៅពេលណាអ្នកជ្រើសរើសយ៉ាងច្បាស់លាស់ដើម្បីនាំចេញរបស់អ្នក។ លំហាត់ប្រាណទៅជាទម្រង់ CSV ហើយចែករំលែកឯកសារដោយខ្លួនឯង។

2. ការអនុញ្ញាតត្រូវបានទាមទារ

2.1 iOS ការអនុញ្ញាត

  • ការចូលប្រើ HealthKit៖អានការចូលប្រើការហាត់ប្រាណ, ចម្ងាយ, ចង្វាក់បេះដូង, ថាមពលសកម្ម, ល្បឿនជិះកង់ និងកម្លាំងជិះកង់
  • បណ្ណាល័យរូបថត (ជាជម្រើស)៖លុះត្រាតែអ្នកជ្រើសរើសរក្សាទុកការសង្ខេបលំហាត់ជារូបភាព

អ្នកអាចគ្រប់គ្រងការអនុញ្ញាត HealthKit គ្រប់ពេលនៅក្នុង iOS ការកំណត់ → ឯកជនភាព និងសុវត្ថិភាព → សុខភាព → កង់ វិភាគ។

2.2 ការអនុញ្ញាត Android

  • android.permission.health.READ_EXERCISE
  • android.permission.health.READ_DISTANCE
  • android.permission.health.READ_HEART_RATE
  • android.permission.health.READ_SPEED
  • android.permission.health.READ_POWER
  • android.permission.health.READ_TOTAL_CALORIES_BURNED
  • ការចូលប្រើអ៊ីនធឺណិត (អ៊ីនធឺណិត): ប្រើសម្រាប់តែការបង្ហាញឋិតិវន្តក្នុងកម្មវិធីប៉ុណ្ណោះ។ មាតិកា និងការចូលប្រើការគ្រប់គ្រងការជាវ។

5.2 ការលុបទិន្នន័យ
